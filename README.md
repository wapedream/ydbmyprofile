        // Элементы DOM
        const cardsContainer = document.getElementById('cardsContainer');
        const addCardBtn = document.getElementById('addCardBtn');
        const cardModal = document.getElementById('cardModal');
        const cardForm = document.getElementById('cardForm');
        const cancelBtn = document.getElementById('cancelBtn');
        
        // Загрузка карточек из localStorage или использование начальных данных
        let cards = JSON.parse(localStorage.getItem('youDreamBabyCards')) || initialCards;

        // Функция для отображения карточек
        function renderCards() {
            cardsContainer.innerHTML = '';
            
            cards.forEach((card, index) => {
                const cardElement = document.createElement('div');
                cardElement.className = 'card';
                cardElement.innerHTML = `
                    <img src="${card.image}" alt="${card.name}" class="card-image">
                    <div class="card-content">
                        <h3 class="card-title">${card.name}</h3>
                        <a href="${card.link}" target="_blank" class="card-link">Перейти <i class="fas fa-external-link-alt"></i></a>
                        <button class="delete-btn" data-index="${index}" style="background: transparent; border: none; color: #ff6b8b; margin-top: 10px; cursor: pointer; font-size: 0.9rem;">
                            <i class="fas fa-trash"></i> Удалить
                        </button>
                    </div>
                `;
                cardsContainer.appendChild(cardElement);
            });
            
            // Добавляем обработчики для кнопок удаления
            document.querySelectorAll('.delete-btn').forEach(btn => {
                btn.addEventListener('click', function() {
                    const index = parseInt(this.getAttribute('data-index'));
                    deleteCard(index);
                });
            });
        }

        // Функция для добавления карточки
        function addCard(name, image, link) {
            cards.push({ name, image, link });
            localStorage.setItem('youDreamBabyCards', JSON.stringify(cards));
            renderCards();
        }

        // Функция для удаления карточки
        function deleteCard(index) {
            if (confirm('Вы уверены, что хотите удалить эту карточку?')) {
                cards.splice(index, 1);
                localStorage.setItem('youDreamBabyCards', JSON.stringify(cards));
                renderCards();
            }
        }

        // Открытие модального окна
        addCardBtn.addEventListener('click', () => {
            cardModal.style.display = 'flex';
            document.getElementById('cardName').focus();
        });

        // Закрытие модального окна
        cancelBtn.addEventListener('click', () => {
            cardModal.style.display = 'none';
            cardForm.reset();
        });

        // Закрытие модального окна при клике вне его
        window.addEventListener('click', (e) => {
            if (e.target === cardModal) {
                cardModal.style.display = 'none';
                cardForm.reset();
            }
        });

        // Обработка отправки формы
        cardForm.addEventListener('submit', (e) => {
            e.preventDefault();
            
            const name = document.getElementById('cardName').value;
            const image = document.getElementById('cardImage').value;
            const link = document.getElementById('cardLink').value;
            
            addCard(name, image, link);
            
            cardModal.style.display = 'none';
            cardForm.reset();
            
            // Прокрутка к новой карточке
            window.scrollTo({
                top: document.body.scrollHeight,
                behavior: 'smooth'
            });
        });

        // Инициализация при загрузке страницы
        document.addEventListener('DOMContentLoaded', () => {
            renderCards();
            
            // Добавляем немного анимации при загрузке
            setTimeout(() => {
                document.querySelectorAll('.card').forEach((card, index) => {
                    card.style.opacity = '0';
                    card.style.transform = 'translateY(20px)';
                    
                    setTimeout(() => {
                        card.style.transition = 'opacity 0.5s ease, transform 0.5s ease';
                        card.style.opacity = '1';
                        card.style.transform = 'translateY(0)';
                    }, index * 100);
                });
            }, 300);
        });
